SELECT I.SUMMARY AS TASK, I2.NAME AS LIST
FROM ISSUES I
JOIN ISSUESLISTS I2 ON I.ISSUESLISTS_ID = I2.ID;

SELECT I.SUMMARY AS TASK, CONCAT(U.FIRSTNAME, ' ', U.LASTNAME) AS USER
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.SUMMARY) AS NUMBER_OF_TASKS
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME
ORDER BY U.LASTNAME, U.FIRSTNAME;
